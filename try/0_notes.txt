   - background-repeat      : no-repeat, repeat-x;
   - display                : inline-block - only takes up as much space as needed;
                              block element - takes up entire line
                              inline element - appear with the line of text
   - direction              : rtl;
   - font-family            : used to change the font of your content;
   - line-height            : space to your paragraph vertically;
   - max-width              : used to limit the wide of ur content;
   - oppacity               : fade 1 to o.1
   - padding                : px, used to expand ur box. top-right bottom-left;
   - text-align             : left, right or auto;
                              : justify, each line is stretched
   - text-indent            : it set space your first paragraph. 50%;
   - vertical-align         : align items to top, middle, bottom
   - width                  : px, vh, %; 
   - word-spacing           : space between word;
